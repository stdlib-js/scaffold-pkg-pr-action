/**
* Computes the squared absolute value of a double-precision floating-point number `x`.
*
* @param {number} x - input value
* @returns {number} squared absolute value
*
* @example
* var v = abs2( -1.0 );
* // returns 1.0
*
* @example

* // returns 4.0

* @example
* var v = abs2( -0.0 );
* // returns 0.0
*

* var v = abs2( NaN );
